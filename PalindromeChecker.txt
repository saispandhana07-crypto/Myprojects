#include <stdio.h>
#include <string.h>
#include <ctype.h>

#define MAX 200

// Function to remove newline from fgets
void stripNewline(char *str) {   
    str[strcspn(str, "\n")] = '\0';
}

// Function to check palindrome sentence
int isPalindrome(char str[]) {  
    char clean[MAX];

    int i, j, len = 0;   

    // Clean the string: keep only letters/numbers, convert to lowercase  
    for (i = 0; str[i]; i++) {    
        if (isalnum(str[i]))  // keep letters and digits
            clean[len++] = tolower(str[i]);    
    }
    clean[len] = '\0';

    // Compare from both ends    
    i = 0;  
    j = len - 1; 

    while (i < j) {
        if (clean[i] != clean[j]) 
            return 0;       // not a palindrome 

        i++;  
        j--; 
    }    

    return 1;     // palindrome
}

int main() {   
    char str[MAX];   

    printf("Enter a sentence: ");  
    fgets(str, MAX, stdin);   
    stripNewline(str); 

    if (isPalindrome(str))  
        printf("It is a Palindrome.\n"); 
    else  
        printf("Not a Palindrome.\n"); 

    return 0;
}
